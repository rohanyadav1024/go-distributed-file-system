1. Root CA

openssl genrsa -out certs/ca/ca.key 4096

openssl req -x509 -new -nodes \
  -key certs/ca/ca.key \
  -sha256 -days 3650 \
  -out certs/ca/ca.crt \
  -subj "/CN=dfs-root-ca"

2. Metad Certificate

openssl genrsa -out certs/metad/server.key 2048

openssl req -new \
  -key certs/metad/server.key \
  -out certs/metad/server.csr \
  -subj "/CN=metad"

echo "subjectAltName=DNS:metad,DNS:localhost,IP:127.0.0.1" > certs/metad/san.ext


3. Storaged Certificate

openssl req -new \
  -key certs/storaged1/server.key \
  -out certs/storaged1/server.csr \
  -subj "/CN=storaged1"

echo "subjectAltName=DNS:storaged1" > certs/storaged1/san.ext

openssl x509 -req \
  -in certs/storaged1/server.csr \
  -CA certs/ca/ca.crt \
  -CAkey certs/ca/ca.key \
  -CAcreateserial \
  -out certs/storaged1/server.crt \
  -days 365 \
  -sha256 \
  -extfile certs/storaged1/san.ext
